var fnUserDbToJs = function (u) {
    // add some properties
    u.id = u.userId;
    u.userGroup = {
        id: u.userGroupId,
        name: u.groupName
    }

    delete u.userId;

    return u;
}

// transforms a js object into a db record
var fnUserJsToDb = function (u) {
    // add properties
    u.userId = u.id;
    // we admit deleted without userGroup
    if (u.userGroup) {
        u.userGroupId = u.userGroup.id;
    }
    // delete some properties
    delete u.id;
    delete u.userGroup;
    return u;
}
